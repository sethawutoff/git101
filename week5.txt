pointer
การบอกโปรแกรมว่าเป็นเลขฐาน 8 คือพิม 0 นำหน้าเลขฐาน 16 คือ 0x 
การใช้ & หน้าตัวแปรเป็นการบอก address ของตัวแปร
* คือ dereferencing ใช้สำหรับการเข้าถึงค่าที่เก็บใน address นั้นๆ
int *ptr , x;   	//กำหนดตัวแปร pointer และตัวแปร x
ptr = &x   		    //ให้ค่าของ ptr เป็น address ของ x
*ptr		        //มีค่าเป็น x

*** ข้อควรระวัง (ห้ามทำ) ***
&ptr = 10;
*ptr = 50;
